<script>\
    let currentStep = 1;\
    const totalSteps = document.querySelectorAll('.step').length;\
    const contentArea = document.querySelector('.content-area');\
\
    document.addEventListener('DOMContentLoaded', () => \{\
        showStep(currentStep);\
    \});\
\
    function showStep(stepNumber) \{\
        const steps = document.querySelectorAll('.step');\
        steps.forEach(step => step.classList.remove('active'));\
        document.querySelector('#step-' + stepNumber).classList.add('active');\
\
        // Update step navigation buttons\
        const stepBtns = document.querySelectorAll('.step-btn');\
        stepBtns.forEach((btn, index) => \{\
            btn.classList.toggle('active', index + 1 === stepNumber);\
        \});\
\
        updateButtons(stepNumber);\
        updateProgressBar(stepNumber);\
        contentArea.scrollTop = 0; // Scroll to top on step change\
    \}\
\
    function goToStep(stepNumber) \{\
        if (stepNumber >= 1 && stepNumber <= totalSteps) \{\
            currentStep = stepNumber;\
            showStep(currentStep);\
        \}\
    \}\
\
    function navigate(direction) \{\
        if ((currentStep > 1 && direction < 0) || (currentStep < totalSteps && direction > 0)) \{\
            currentStep += direction;\
            showStep(currentStep);\
        \}\
    \}\
\
    function updateButtons(stepNumber) \{\
        document.getElementById('prevBtn').disabled = (stepNumber === 1);\
        document.getElementById('nextBtn').disabled = (stepNumber === totalSteps);\
    \}\
\
    function updateProgressBar(stepNumber) \{\
        const progress = document.getElementById('progress');\
        const progressPercentage = ((stepNumber - 1) / (totalSteps - 1)) * 100;\
        progress.style.width = progressPercentage + '%';\
        progress.innerText = 'Step ' + stepNumber + ' of ' + totalSteps;\
    \}\
